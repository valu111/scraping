      container.innerHTML = "";

        container.innerHTML = `
          <div class="card">
            <img src="potaxie.jpg" alt="Sin resultados">

        `;

      }
      lista.forEach(j => {
        container.innerHTML += `
          <div class="card">
            <img src="${j.imagen}" alt="Imagen del empleo">
            <h1>T√≠tulo: ${j.titulo}</h1>
            <h3>Subt√≠tulo: ${j.subtitulo}</h3>
            <p>Texto: ${j.texto}</p>
